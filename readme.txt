git init                         # 1. Initialize a new local Git repository
git add .                        # 2. Stage all changes
git commit -m "Initial commit"  # 3. Commit the changes with a message
git branch -M main              # 4. Rename the default branch to 'main'
git remote add origin <repository-URL>   # 5. Link your local repo to the remote (GitHub)
git push -u origin main         # 6. Push your changes to the 'main' branch on GitHub
